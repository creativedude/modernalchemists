NodeMCU - ESP8266

  • NodeMCU is a development board similar to that of the Arduino based on the popular ESP8266 WIFI SOC
  • It provides support for I2C, UART, SPI, Analog IO, Digital IO, PWM & interrupt
  • Pre-Programmed with Lua interpreter
  • Can be programmed in Lua via ESPlorer IDE or in C/C++ within the Arduino IDE
  • We will be looking at programming the ESP/NodeMCU in the Arduino IDE

API & Rest

  • REST (REpresentational State Transfer) is the underlying architectural principle that provides communication between devices on the internet.
  • Browsers use rest calls to retrieve or submit data to the server.
  • Server Exposes an API (Application Programming Interface)
  • HTTP verbs: GET, POST, PUT, DELETE, OPTIONS, etc..
  • Almost every largescale system uses rest

MQTT

  • Message Queue Telemetry Transport
  • M2M (Machine-to-machine)/ IoT
  • Extremely Lightweight
  • Queueing of messages with broker distributing them by topic

ESP8266 - API

Lets build a web server that hosts an API so that we can control an LED.

The code below sets up an API at '<IP_ADDRESS>/data' that takes in a json object with one property. Property 'LED' can be set to either 'on' or 'off' corresponding to the state of teh built in LED. 

//Import Libraries
#include <ArduinoJson.h>
#include <ESP8266WiFi.h>
#include <WiFiClient.h>
#include <ESP8266WebServer.h>
#include <ESP8266mDNS.h>

//const char *ssid = "Tell-my-wifi-love-her";
//const char *password = "B6CDF49007";


//Define Global Variables
ESP8266WebServer server ( 80 );
String _ledState = "off";

//Set Wifi Credentials
const char *ssid = "ModernAlchemists";
const char *password = "I_Like_Pizza";

//Home Page HTML
String HTML_Root = "<!doctype html> <html> <head> ESP8266 Api</head> <body>\
<h1>WELCOME!</h1>\
<p>To turn the LED on/off send 'POST' with body {\"LED\":\"**STATE**\"} to **IP**/data</p>\
</body> </html>";

//API Response
String HTML_Api = "<!doctype html> <html> <head> ESP8266 Web Server </head> <body>\
<p>The LED is: </p>\
</body> </html>";

//Root Callback Function
void handleRoot() {
  server.send ( 200, "text/html", HTML_Root );
}

//API Callback Function
void handlePost(){
  //Get & Deserialize request
  StaticJsonBuffer<350> jsonBuffer;
  JsonObject& root = jsonBuffer.parseObject(server.arg("plain"));
  String led = root["LED"];

  //Change LED State
  Serial.println("Turning LED: " + led);
  if (led == "on") 
  {
    digitalWrite(LED_BUILTIN, LOW);
  }
  else if (led == "off")
  {
    digitalWrite(LED_BUILTIN, HIGH); 
  }
   _ledState = led;
  server.send ( 200, "text/html", HTML_Api);
}

//Page Not Found Callback
void handleNotFound() {
  server.send ( 404, "text/plain", "Error! Not Found\n\n" );
}



/* SETUP
 * 1) Initialize Serial
 * 2) Initialize LED
 * 3) Connect to Wifi
 */
void setup() {
  // Define & Set LED
  pinMode(LED_BUILTIN, OUTPUT);     
  digitalWrite(LED_BUILTIN, HIGH); 

  //Initialize Serial & baud rate
  Serial.begin ( 115200 );

  //Initialize Wifi
  WiFi.begin ( ssid, password );

  // Wait for connection & Print status
  while ( WiFi.status() != WL_CONNECTED ) {
      delay ( 500 );
      Serial.print ( "." );
    }

  //Map Routes
  server.on ( "/", handleRoot );
  server.on ( "/data", handlePost);
  server.onNotFound ( handleNotFound );
  //Begin Server
  server.begin();

  //Print Connection
  Serial.println ( "HTTP server started\n" );
  Serial.print ( "Connected to " );
  Serial.println ( ssid );
  Serial.print ( "IP address: " );
  Serial.println ( WiFi.localIP() );
}




/*
 * MAIN LOOP 
 */
void loop() {
    server.handleClient();
}

MQTT Example

  • Lets do the same thing as before with REST, but this time with MQTT
  • So we will be switching on and off the LED by listening to a topic
  • Also, when the server says hello to the devices, lets say hello back.
#include <ESP8266WiFi.h>
#include <PubSubClient.h>

//Define Client
WiFiClient espClient;
PubSubClient client(espClient);

//Configure credentials
const char* ssid = "ModernAlchemists";
const char* password = "I_Like_Pizza";

//MQTT Settings
const char* mqtt_server = "192.168.0.100";    //IP address of Raspberry pi running mosquitto
const char* inTopic = "ModernAlchemists";

//Callback for when message is received
void callback(char* topic, byte* payload, unsigned int length) {
  Serial.print("TOPIC: ");
  Serial.println(topic);

  const char x = ((char)payload[0]);
  Serial.print("MSG: ");
  Serial.println(x);
  
  // Switch on the LED if an 1 was received as first character
  if ( x == '1') {
    digitalWrite(LED_BUILTIN, LOW);   // Turn the LED on (Note that LOW is the voltage level
    Serial.println("LED: On");
  } else if (x =='0') {
    digitalWrite(LED_BUILTIN, HIGH); // Turn the LED off by making the voltage HIGH
    Serial.println("LED: Off");
  } 
  else if (x == 'H'){
   bool published = client.publish("HelloFromESP", "Hello from ESP");   
    }
  Serial.println(""); 
  
}

/*
  * Initialize Pubsub client and wifi client.
  * Define MQTT broker connection on port 1883.
  * Define callback methods for MQTT event requests.
  * 
*/
void setup() {
  pinMode(LED_BUILTIN, OUTPUT);
  digitalWrite(LED_BUILTIN, HIGH);  // Turn the LED off by making the voltage HIGH
  Serial.begin(115200);
  delay(10);


  // We start by connecting to a WiFi network
  Serial.print("Connecting to ");
  Serial.println(ssid);
  WiFi.begin(ssid, password);
  
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  Serial.println("");
  Serial.println("WiFi connected");  
  Serial.println("IP address: ");
  Serial.println(WiFi.localIP());

 
  client.setServer(mqtt_server, 1883);
  client.setCallback(callback);
}



void reconnect() {
 // Loop until we're reconnected
 while (!client.connected()) {
 Serial.println("Attempting to reconnect.");
 // Attempt to connect
 if (client.connect("ESP8266 Client")) {
  Serial.println("connected!");
  //Subscribe to topic once connection is established.
  client.subscribe(inTopic);
 } else {
  Serial.print("failed, rc=");
  Serial.print(client.state());
  Serial.println("trying again");
  delay(5000);
  }
 }
}

/*
 * Attempt reconnection until connection is established. 
 * In reconnect function bind to topic.
*/
void loop() {
if (!client.connected()) {
  reconnect();
 }
 client.subscribe(inTopic);
 client.loop();
}
